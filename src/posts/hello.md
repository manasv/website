I'll drop a fibonacci recursive code in Swift just for testing the syntax highlighting:

```swift

func fibonacci(number: Int) -> Int {
    
    if number == 0 {
        return 0
    } else if number == 1 {
        return 1
    }
    
    return fibonacci(number - 1) + fibonacci(number - 2)

}
```
